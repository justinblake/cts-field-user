import { Injectable } from '@angular/core';
import { Http } from '@angular/http';
import 'rxjs/add/operator/map';
import {ApiService, ApiData} from './api-service';
//import {StorageService} from './storage-service';
 
 
@Injectable()
export class TaskManager {
  tmpData:any;
  useLocalData:boolean = true;
  currentUser:any;
  currentTask:any;
  constructor(public apiService: ApiService /*, storageService: StorageService*/) {
    console.log('Hello TaskManager Provider');
    this.tmpData = new ApiData();
  }
 
  // workaround put in place when CORS wasn't setup on remote server
  getCurrentTaskLocal(){
    return new Promise( (resolve, reject)=> {
      this.currentUser = this.tmpData.data.authenticateUser;
      this.currentTask = this.tmpData.data.loadCurrentTask;
 
      let response = {
        userdata : this.tmpData.data.authenticateUser.userdata,
        task : this.tmpData.data.loadCurrentTask
      }
 
      resolve(response);
    });
  }
 
  // get current task from remote api
   getCurrentTaskRemote(){
 
    return new Promise( (resolve, reject)=> {
      this.apiService.authenticate({}).then(user => {
        console.log('TASK MGR user: ' + user);
        return this.apiService.loadCurrentTask(user);
      }).then(task => {
        console.log('TASK MGR task: ' + task);
        resolve(task);
      }).catch(error => {
        //TODO
      })
    });
  };
 
  getCurrentTask(){
    if(this.useLocalData === true){
      return this.getCurrentTaskLocal();
    }else{
      return this.getCurrentTaskRemote();
    }
  }
  getTaskHistory(){
    return new Promise( (resolve, reject)=> {
      //resolve(this.tmpData.data.loadTaskHistory);
      this.apiService.loadTaskHistory({}).then(json => {
        resolve(json);
      }).catch(error => {
        //TODO
      })
    });
  }
  authenticateUser(){
    return new Promise( (resolve, reject)=> {
      //resolve(this.tmpData.data.authenticateUser);
      this.apiService.authenticate({}).then(json => {
        resolve(json);
      }).catch(error => {
        //TODO
      })
    });
  }
 
}
